### Sending Email with Attachment
```python
from django.core.mail import EmailMessage
from django.template.loader import render_to_string

def send_invoice_email(invoice, email_type='paid'):
    # Render email template
    context = {
        'athlete_name': invoice.customer_name.split()[0],
        'invoice_number': invoice.invoice_number,
        'amount': invoice.total_amount,
        'payment_date': invoice.invoice_date,
        'service': invoice.payment.subscription.billing_plan.name,
        'period': invoice.payment.months_covered,
        'due_date': invoice.due_date,
        'upi_id': invoice.template.bank_upi_id,
    }
    
    # Select template based on type
    template_map = {
        'paid': 'emails/invoice_paid.html',
        'unpaid': 'emails/invoice_unpaid.html',
        'overdue': 'emails/invoice_overdue.html',
        'cancelled': 'emails/invoice_cancelled.html',
    }
    
    subject_map = {
        'paid': f'Payment Received - Invoice {invoice.invoice_number}',
        'unpaid': f'Invoice for {context["period"]} - Payment Due',
        'overdue': f'‚ö†Ô∏è Payment Overdue - Invoice {invoice.invoice_number}',
        'cancelled': f'Invoice Cancelled - {invoice.invoice_number}',
    }
    
    html_message = render_to_string(template_map[email_type], context)
    
    # Create email
    email = EmailMessage(
        subject=subject_map[email_type],
        body=html_message,
        from_email=settings.DEFAULT_FROM_EMAIL,
        to=[invoice.customer_email],
    )
    email.content_subtype = 'html'
    
    # Attach PDF
    if invoice.pdf_file:
        email.attach_file(invoice.pdf_file.path)
    
    # Send
    email.send()
    
    # Update invoice
    invoice.emailed_at = timezone.now()
    invoice.save()
```
